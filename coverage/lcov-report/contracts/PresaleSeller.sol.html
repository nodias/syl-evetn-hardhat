pragma solidity ^0.5.6;
 
import "./klaytn-contracts/math/SafeMath.sol";
import "./klaytn-contracts/token/KIP7/KIP7.sol";
import "./klaytn-contracts/ownership/Ownable.sol";
import "hardhat/console.sol";
 
contract PresaleSeller is Ownable {
    using SafeMath for uint256;
 
    KIP7 token;
    uint256 maxBuy;
    uint256 totalBought;
    mapping(address => uint256) buyers;
    mapping(address => uint256) public whitelists;
    uint256 price;
    address payable private receiver;
 
    enum WHITELIST {
        NOT,
        IS,
        EARLY
    }
 
    // address : 토큰 컨트랙트 주소
    // maxBuy : 하드캡
    // _receiver : 클레이 받을 주소
    // _price : 토큰 가격 (1 클레이당 몇개를 주는가)
    constructor(
        address _address,
        uint256 _maxBuy,
        address payable _receiver,
        uint256 _price
    ) public {
        token = KIP7(_address);
        maxBuy = _maxBuy;
        receiver = _receiver;
        price = _price;
        totalBought = 0;
    }
 
    function changeWhiteList(address toAdd, uint256 status) external onlyOwner {
        whitelists[toAdd] = status;
    }
 
    function bought() public view returns (uint256) {
        return buyers[msg.sender];
    }
 
    function getTotalBought() public view returns (uint256) {
        return totalBought;
    }
 
    function setPrice(uint256 _price) public onlyOwner {
        price = _price;
    }
 
    function buy() public payable {
        require(msg.value != 0, "msg.value == 0");
        require(
            whitelists[msg.sender] > uint256(WHITELIST.NOT),
            "not whiteList"
        );
 
        uint256 buyAmount = (msg.value).div(price.mul(1e15)) * 1e18;
        require(
            buyAmount > 0,
            "The entered value is lower than the token price."
        );
        require(
            buyAmount + totalBought < maxBuy,
            "Exceeding the maximum number of purchases"
        );
 
        require(
            buyers[msg.sender] + buyAmount <=
                whitelists[msg.sender] * 100000 * 1e18,
            "Exceeding the number of purchases per person"
        );
 
        buyers[msg.sender] += buyAmount;
        totalBought += buyAmount;
        token.transfer(msg.sender, buyAmount);
        address(receiver).transfer(msg.value);
    }
}
