pragma solidity ^0.5.6;
 
import "./klaytn-contracts/math/SafeMath.sol";
import "./klaytn-contracts/token/KIP7/IKIP7.sol";
import "./interfaces/IPool.sol";
import "./Brag.sol";
import "hardhat/console.sol";
 
contract BragPool is IPool {
    using SafeMath for uint256;
 
    IKIP7 public brag;
 
    constructor(IKIP7 _Brag) public {
        brag = _Brag;
    }
 
    function() external payable {}
 
    function swapToBrag() external payable {
        uint256 newKlay = address(this).balance;
        uint256 lastBrag = brag.balanceOf(address(this));
 
        uint256 newBrag = (newKlay.sub(msg.value)).mul(lastBrag).div(newKlay);
 
        brag.transfer(msg.sender, lastBrag.sub(newBrag));
 
        emit SwapToBrag(msg.sender, msg.value);
    }
 
    function swapToKlay(uint256 amount) external {
        uint256 lastKlay = address(this).balance;
        uint256 lastBrag = brag.balanceOf(address(this));
 
        uint256 newKlay = lastBrag.mul(lastKlay).div(
            lastBrag.add(amount.mul(9).div(10))
        );
 
        brag.transferFrom(msg.sender, address(this), amount);
        msg.sender.transfer(lastKlay.sub(newKlay));
 
        emit SwapToKlay(msg.sender, amount);
    }
 
    function addLiquidity() external payable {
        console.log("IK : %s" , msg.value);
        console.log("NK : %s" , address(this).balance);
        uint256 lastKlay = (address(this).balance).sub(msg.value);
        console.log("LK : %s" , lastKlay);
        uint256 lastBrag = brag.balanceOf(address(this));
        console.log("LB : %s" , lastBrag);
 
        uint256 inputBrag = lastBrag.mul(msg.value).div(address(this).balance);
        console.log(brag.transferFrom(msg.sender, address(this), inputBrag));
        console.log("NB : %s" , brag.balanceOf(address(this)));
    }
}
